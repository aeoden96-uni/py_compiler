VRSTE VARIJABLI =>  LOG_VAR i DEF_VAR



START 			-> NAREDBA VIŠE
VIŠE  			-> NAREDBA VIŠE | ''

NAREDBA 		-> 	PETLJA |
					GRANANJE |
					PIŠI ; |
					BREAK ; | 
					PRIDRUZIVANJE ;|
					KOMPREHENZIJA ;

PETLJA 			-> FOR NAREDBA | FOR { VIŠE } | 
				   FOREACH NAREDBA | FOREACH { VIŠE }

FOR     		-> for( DEF_VAR# =  BROJ ; DEF_VAR# < BROJ ; DEF_VAR# INCR)
FOREACH 		-> foreach DEF_VAR# in LISTA

INCR 			-> ++ | +BROJ

#KOMPREHENZIJA 	->  DEF_VAR OPER_PRIDRUZ [ aritm_operacija FOREACH ] |
#				  [ izraz FOREACH ]	

PIŠI 			->  #U_OKOLINU ( izraz ) 

ČITAJ 			-> $IZ_OKOLINE( ) | $IZ_OKOLINE ( BOOL )

GRANANJE 		-> 	if( LOGIKA ) { VIŠE }<- jos ne zagrade

PRIDRUZIVANJE 	->	DEF_VAR OPER_PRIDRUZ    DEF_VAR |               <-visak?    
					DEF_VAR OPER_PRIDRUZ    izraz   |
					DEF_VAR OPER_PRIDRUZ    $fja   |
					LOG_VAR OPER_PRIDRUZ 	LOGIKA    <- moze biti tu i read fja
				 
		 		 
LOGIKA 			->	KONJ ILI LOGIKA | KONJ
KONJ  			->  log_literal I KONJ      | log_literal
log_literal 	->  izraz <  izraz | 
					izraz == izraz |
					( LOGIKA )     |
					T | F | U  | 
					LOG_VAR |
					not LOGIKA |

faktor 			-> 	baza NA faktor | baza 
baza 			-> 	BROJ | 	
					(tip)izraz | 
					$IZ_OKOLINE( ) |

					( izraz )
